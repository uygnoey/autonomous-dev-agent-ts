# Layer3 설계: types.ts

## 1. 개요

**목적**: Layer3 전용 타입 정의 (문서 템플릿, 산출물 타입 등)

**위치**: `src/layer3/types.ts`

**의존성**: layer3 → core

**핵심 책임**:
- 문서 템플릿 타입 정의 (프로젝트 문서 8개, 비즈니스 산출물 4개)
- 산출물 빌더 타입 정의
- 버그 리포트 타입 정의
- 지속 E2E 실행 설정 타입

---

## 2. 타입 정의

```typescript
/**
 * layer3 (3계층) 타입 정의 / Layer 3 type definitions
 *
 * @description
 * KR: 산출물 생성, 지속 검증에 사용되는 타입.
 *     문서 템플릿, 산출물 타입, 버그 리포트, 지속 E2E 설정 등.
 * EN: Types for deliverable generation and continuous verification.
 *     Document templates, deliverable types, bug reports, continuous E2E config.
 */

import type { Phase } from '../core/types.js';

// ── 문서 템플릿 / Document Template ─────────────────────────────

/**
 * 프로젝트 문서 템플릿 타입 / Project document template type
 *
 * @description
 * KR: 3계층에서 생성하는 프로젝트 문서 8개 유형.
 * EN: 8 types of project documents generated in Layer 3.
 */
export type ProjectDocumentType =
  | 'readme'
  | 'api-reference'
  | 'architecture'
  | 'user-manual'
  | 'installation-guide'
  | 'test-report'
  | 'changelog'
  | 'contributing-guide';

/**
 * 비즈니스 산출물 타입 / Business deliverable type
 *
 * @description
 * KR: 3계층에서 생성하는 비즈니스 산출물 4개 유형.
 * EN: 4 types of business deliverables generated in Layer 3.
 */
export type BusinessDeliverableType =
  | 'portfolio'
  | 'business-plan'
  | 'investment-proposal'
  | 'presentation';

/**
 * 문서 템플릿 / Document template
 *
 * @description
 * KR: 문서 생성에 사용되는 템플릿 정의.
 * EN: Template definition used for document generation.
 */
export interface DocumentTemplate {
  /** 템플릿 ID / Template ID */
  readonly id: string;
  /** 템플릿 이름 / Template name */
  readonly name: string;
  /** 문서 유형 / Document type */
  readonly type: ProjectDocumentType | BusinessDeliverableType;
  /** 템플릿 파일 경로 / Template file path */
  readonly templatePath: string;
  /** 출력 형식 / Output format */
  readonly format: 'md' | 'html' | 'pdf' | 'docx' | 'pptx';
  /** 템플릿 설명 / Template description */
  readonly description: string;
  /** 커스텀 템플릿 여부 / Whether custom template */
  readonly custom: boolean;
}

/**
 * 문서 조각 / Document fragment
 *
 * @description
 * KR: 2계층 documenter가 생성한 개별 조각 문서.
 * EN: Individual fragment document generated by Layer 2 documenter.
 */
export interface DocumentFragment {
  /** 조각 ID / Fragment ID */
  readonly id: string;
  /** 기능 ID / Feature ID */
  readonly featureId: string;
  /** 조각 유형 / Fragment type */
  readonly type: 'feature-doc' | 'test-result' | 'api-spec' | 'bug-report' | 'changelog' | 'review' | 'decision';
  /** 조각 내용 / Fragment content */
  readonly content: string;
  /** 생성 시각 / Created at */
  readonly createdAt: Date;
  /** 메타데이터 / Metadata */
  readonly metadata: Readonly<Record<string, unknown>>;
}

// ── 통합 문서 / Integrated Document ─────────────────────────────

/**
 * 통합 문서 / Integrated document
 *
 * @description
 * KR: 여러 조각을 통합하여 생성된 최종 문서.
 * EN: Final document generated by integrating multiple fragments.
 */
export interface IntegratedDocument {
  /** 문서 ID / Document ID */
  readonly id: string;
  /** 프로젝트 ID / Project ID */
  readonly projectId: string;
  /** 문서 유형 / Document type */
  readonly type: ProjectDocumentType;
  /** 문서 내용 / Document content */
  readonly content: string;
  /** 출력 파일 경로 / Output file path */
  readonly outputPath: string;
  /** 사용된 조각 ID 목록 / Used fragment IDs */
  readonly fragmentIds: readonly string[];
  /** 생성 시각 / Generated at */
  readonly generatedAt: Date;
}

// ── 협업 문서 / Collaborative Document ──────────────────────────

/**
 * 협업 문서 생성 옵션 / Collaborative document generation options
 *
 * @description
 * KR: 1계층(뼈대) + 2계층(상세) 협업 문서 생성 옵션.
 * EN: Options for collaborative document generation between Layer 1 (structure) and Layer 2 (details).
 */
export interface CollaborativeDocOptions {
  /** 프로젝트 ID / Project ID */
  readonly projectId: string;
  /** 문서 유형 / Document type */
  readonly type: ProjectDocumentType | BusinessDeliverableType;
  /** 1계층 뼈대 내용 / Layer 1 structure content */
  readonly layer1Structure: string;
  /** 2계층 조각 목록 / Layer 2 fragment list */
  readonly layer2Fragments: readonly DocumentFragment[];
  /** 출력 경로 / Output path */
  readonly outputPath: string;
}

/**
 * 협업 문서 결과 / Collaborative document result
 */
export interface CollaborativeDocResult {
  /** 문서 ID / Document ID */
  readonly id: string;
  /** 최종 내용 / Final content */
  readonly content: string;
  /** 출력 파일 경로 / Output file path */
  readonly outputPath: string;
  /** 생성 시각 / Generated at */
  readonly generatedAt: Date;
}

// ── 지속 E2E 검증 / Continuous E2E Verification ─────────────────

/**
 * 지속 E2E 실행 설정 / Continuous E2E execution config
 *
 * @description
 * KR: 3계층에서 문서 생성과 병행하여 실행하는 지속 E2E 설정.
 * EN: Config for continuous E2E execution running in parallel with document generation in Layer 3.
 */
export interface ContinuousE2EConfig {
  /** 프로젝트 ID / Project ID */
  readonly projectId: string;
  /** E2E 테스트 경로 / E2E test path */
  readonly testPath: string;
  /** 실행 간격 (ms) / Execution interval in milliseconds */
  readonly intervalMs: number;
  /** Fail-Fast 활성화 / Enable fail-fast */
  readonly failFast: boolean;
  /** 최대 동시 실행 수 / Max concurrent executions */
  readonly maxConcurrency: number;
}

/**
 * 지속 E2E 실행 결과 / Continuous E2E execution result
 */
export interface ContinuousE2EResult {
  /** 실행 ID / Execution ID */
  readonly id: string;
  /** 프로젝트 ID / Project ID */
  readonly projectId: string;
  /** 실행 시각 / Executed at */
  readonly executedAt: Date;
  /** 통과 여부 / Whether passed */
  readonly passed: boolean;
  /** 실패한 테스트 (있는 경우) / Failed test (if any) */
  readonly failedTest?: string;
  /** 에러 메시지 (실패 시) / Error message (if failed) */
  readonly errorMessage?: string;
}

// ── 버그 리포트 / Bug Report ─────────────────────────────────────

/**
 * 버그 심각도 / Bug severity
 */
export type BugSeverity = 'critical' | 'high' | 'medium' | 'low';

/**
 * 버그 카테고리 / Bug category
 */
export type BugCategory = 'design-flaw' | 'implementation-bug' | 'test-gap' | 'regression';

/**
 * 버그 리포트 / Bug report
 *
 * @description
 * KR: 3계층 지속 E2E에서 발견한 버그를 2계층에 보고하는 리포트.
 * EN: Bug report sent from Layer 3 continuous E2E to Layer 2.
 */
export interface BugReport {
  /** 리포트 ID / Report ID */
  readonly id: string;
  /** 프로젝트 ID / Project ID */
  readonly projectId: string;
  /** 기능 ID (연관된 경우) / Feature ID (if related) */
  readonly featureId?: string;
  /** 버그 제목 / Bug title */
  readonly title: string;
  /** 버그 설명 / Bug description */
  readonly description: string;
  /** 재현 단계 / Reproduction steps */
  readonly reproductionSteps: readonly string[];
  /** 기대 동작 / Expected behavior */
  readonly expectedBehavior: string;
  /** 실제 동작 / Actual behavior */
  readonly actualBehavior: string;
  /** 심각도 / Severity */
  readonly severity: BugSeverity;
  /** 카테고리 / Category */
  readonly category: BugCategory;
  /** 근본 원인 (qc 분석 결과) / Root cause (from qc analysis) */
  readonly rootCause?: string;
  /** 리포트 생성 시각 / Reported at */
  readonly reportedAt: Date;
}

/**
 * 버그 에스컬레이션 결과 / Bug escalation result
 *
 * @description
 * KR: 버그를 2계층에 에스컬레이션한 결과.
 * EN: Result of escalating a bug to Layer 2.
 */
export interface BugEscalationResult {
  /** 에스컬레이션 ID / Escalation ID */
  readonly id: string;
  /** 버그 리포트 ID / Bug report ID */
  readonly bugReportId: string;
  /** 2계층 재실행 트리거됨 / Layer 2 re-execution triggered */
  readonly triggered: boolean;
  /** 대상 Phase (architect부터 시작) / Target phase (starting from architect) */
  readonly targetPhase: Phase;
  /** 에스컬레이션 시각 / Escalated at */
  readonly escalatedAt: Date;
}

// ── 비즈니스 산출물 / Business Deliverable ───────────────────────

/**
 * 비즈니스 산출물 메타데이터 / Business deliverable metadata
 */
export interface DeliverableMetadata {
  /** 프로젝트 이름 / Project name */
  readonly projectName: string;
  /** 프로젝트 설명 / Project description */
  readonly projectDescription: string;
  /** 대상 독자 / Target audience */
  readonly targetAudience: string;
  /** 생성 목적 / Purpose */
  readonly purpose: string;
  /** 추가 메타데이터 / Additional metadata */
  readonly extra: Readonly<Record<string, unknown>>;
}

/**
 * 비즈니스 산출물 빌드 옵션 / Business deliverable build options
 */
export interface DeliverableBuildOptions {
  /** 프로젝트 ID / Project ID */
  readonly projectId: string;
  /** 산출물 유형 / Deliverable type */
  readonly type: BusinessDeliverableType;
  /** 템플릿 ID / Template ID */
  readonly templateId: string;
  /** 메타데이터 / Metadata */
  readonly metadata: DeliverableMetadata;
  /** 출력 경로 / Output path */
  readonly outputPath: string;
}

/**
 * 비즈니스 산출물 / Business deliverable
 */
export interface BusinessDeliverable {
  /** 산출물 ID / Deliverable ID */
  readonly id: string;
  /** 프로젝트 ID / Project ID */
  readonly projectId: string;
  /** 산출물 유형 / Deliverable type */
  readonly type: BusinessDeliverableType;
  /** 출력 파일 경로 / Output file path */
  readonly outputPath: string;
  /** 생성 시각 / Generated at */
  readonly generatedAt: Date;
}
```
